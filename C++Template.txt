int N=100000000;
bool isprime[100000000];
void prime(){  //Sieve of Eratosthenes(Sieve Algorithm) 
   for(int i=2; i<=N; i++){
      isprime[i] = true;
   }
   for(int i=2; i*i<=N; i++){
      if(isprime[i]==true){
         for(int j=i*i; j<=N; j+=i){
            isprime[j] = false;
         }
      }
   }
} // Time limit for this algorithm is - O(N(log(logN)))
